Fill the missing data with ramdom numbers

set.seed(10)
z  <- floor(runif(nrow(activity), 
                  min = min(activity$steps, na.rm = T), 
                  max = max(activity$steps, na.rm = T)/10))
w <- which(is.na(activity$steps))
activity$steps[w] <- z[w]
